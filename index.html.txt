<script>
  const code = "NT9Y2ZJX";
  const codeBox = document.getElementById('codeBox');
  const copyBtn = document.getElementById('copyBtn');
  const hint = document.getElementById('hint');

  async function copyCode(){
    try{
      await navigator.clipboard.writeText(code);
    }catch(e){
      const ta = document.createElement('textarea');
      ta.value = code;
      ta.style.position = 'fixed';
      ta.style.opacity = '0';
      document.body.appendChild(ta);
      ta.select();
      document.execCommand('copy');
      document.body.removeChild(ta);
    }
    codeBox.classList.add('copied');
    copyBtn.classList.add('copied');
    copyBtn.textContent = 'Copied ✓';
    hint.classList.add('success');
    hint.innerHTML = 'Copied — ready to paste at checkout';

    setTimeout(()=>{
      codeBox.classList.remove('copied');
      copyBtn.classList.remove('copied');
      copyBtn.textContent = 'Copy Code';
      hint.classList.remove('success');
      hint.innerHTML = '<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="9" y="9" width="11" height="11" rx="2"/><path d="M5 15H4a2 2 0 0 1-2-2V4a2 2 0 0 1 2-2h9a2 2 0 0 1 2 2v1"/></svg> Tap the code or button to copy';
    }, 2200);
  }

  codeBox.addEventListener('click', copyCode);
  codeBox.addEventListener('keypress', (e)=>{ if(e.key==='Enter' || e.key===' ') copyCode(); });
  copyBtn.addEventListener('click', copyCode);
</script>
